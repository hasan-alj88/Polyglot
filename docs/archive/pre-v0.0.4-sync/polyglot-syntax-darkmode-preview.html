[|] ProcessDataWithRecovery
[i] .raw_data: pg\string
[t] |T.Call
[W] |W.NoSetup.NoCleanup

// Sequential execution: Step 1 - Validate input
[r] |ValidateInput
[<] .data: pg\string << .raw_data
[>] .validated: pg\string >> .clean_data
[~]
[~][!] !ValidationError
[~][>] .message: pg\string >> .err_msg
[~][>] .code: pg\int >> .err_code
[~]
[~]// Recovery: Use default value
[~][r] .clean_data: pg\string << "DEFAULT_VALUE"

[o] .output: pg\string << .clean_data
[X]


[|] ParallelGreetings
[i] .name: pg\string
[t] |T.Call
[W] |W.NoSetup.NoCleanup

[r] .english_greeting: pg\string << ""
[r] .spanish_greeting: pg\string << ""
[r] .french_greeting: pg\string << ""

// Parallel execution: Generate greetings concurrently
[p] |FormatEnglish
[<] .name: pg\string << .name
[>] .greeting: pg\string >> .english_greeting

[~]

[p] |FormatSpanish
[<] .name: pg\string << .name
[>] .greeting: pg\string >> .spanish_greeting

[~]

[p] |FormatFrench
[<] .name: pg\string << .name
[>] .greeting: pg\string >> .french_greeting

// Join: Wait for all parallel operations to complete
[Y] |Y.Join

// Boolean logic: Check if all greetings succeeded
[&] .english_greeting ?> ""
[&] .spanish_greeting ?> ""
[&] .french_greeting ?> ""
[~][r] .all_success: pg\bool << True

[r] .combined: pg\string << "{.english_greeting}, {.spanish_greeting}, {.french_greeting}"
[o] .greetings: pg\string << .combined
[X]
